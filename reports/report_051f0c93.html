WITH SalesData AS (
    SELECT
        product_name,
        region,
        city,
        total_amount,
        CASE
            WHEN unit_price = 0 THEN 0
            ELSE total_amount / unit_price
        END AS quantity_sold
    FROM
        fact_sales
    WHERE
        total_amount > 0
),
OverallMetrics AS (
    SELECT
        SUM(total_amount) AS total_sales,
        SUM(quantity_sold) AS total_quantity_sold,
        AVG(total_amount) AS average_transaction_value
    FROM
        SalesData
),
TopProductsByAmount AS (
    SELECT
        product_name,
        SUM(total_amount) AS product_total_amount
    FROM
        SalesData
    GROUP BY
        product_name
    ORDER BY
        product_total_amount DESC
    LIMIT 5
),
TopProductsByQuantity AS (
    SELECT
        product_name,
        SUM(quantity_sold) AS product_total_quantity
    FROM
        SalesData
    GROUP BY
        product_name
    ORDER BY
        product_total_quantity DESC
    LIMIT 5
),
SalesByRegion AS (
    SELECT
        region,
        SUM(total_amount) AS region_sales
    FROM
        SalesData
    GROUP BY
        region
    ORDER BY
        region_sales DESC
),
SalesByCity AS (
    SELECT
        city,
        SUM(total_amount) AS city_sales
    FROM
        SalesData
    GROUP BY
        city
    ORDER BY
        city_sales DESC
)
SELECT
    (SELECT total_sales FROM OverallMetrics) AS total_sales,
    (SELECT total_quantity_sold FROM OverallMetrics) AS total_quantity_sold,
    (SELECT average_transaction_value FROM OverallMetrics) AS average_transaction_value,
    (SELECT json_group_array(json_object('product_name', product_name, 'product_total_amount', product_total_amount)) FROM TopProductsByAmount) AS top_products_by_amount,
    (SELECT json_group_array(json_object('product_name', product_name, 'product_total_quantity', product_total_quantity)) FROM TopProductsByQuantity) AS top_products_by_quantity,
    (SELECT json_group_array(json_object('region', region, 'region_sales', region_sales)) FROM SalesByRegion) AS sales_by_region,
    (SELECT json_group_array(json_object('city', city, 'city_sales', city_sales)) FROM SalesByCity) AS sales_by_city
